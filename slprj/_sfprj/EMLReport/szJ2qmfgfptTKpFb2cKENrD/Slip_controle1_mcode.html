1function [Trl_max,Trr_max] = Slip_controle(ax,ay,mu_max,Vzwp)
  2% Berekening van de maximale torques die men kan toepassen op de wielen
  3% door berekening van de normaalkrachten op de wielen.
  4
  5%load('parameters.mat');
  6
  7M       =265;               
  8L       =1.525;             
  9Lf      =0.780;             
 10Lr      =0.745;             
 11Tf      =1.200;             
 12Tr      =1.200;             
 13g       =9.81;              
 14Ksf     =12153;             
 15Ksr     =14563;             
 16Krf_a   =183.26;            
 17Krr_a   =325.31;            
 18Kwf     =0;                 
 19Kwr     =0;                 
 20Hrcf    =0.032;             
 21Hrcr    =0.041;             
 22Hurcf   =0;                 
 23Hurcr   =0;                 
 24H       =0.2735;            
 25D       =0.2369;            
 26Rm      =576.35;            
 27Reff    =0.2325;            
 28Gr      =10.32;             
 29Sr      =0.600;             
 30Tmax    =38.5;              
 31Cte     =0;                 
 32%ax      =0;                 
 33%ay      =0;                 
 34Mu_f    =8.5;               
 35Mu_r    =8.5;               
 36R1      =0;                 
 37R2      =0;                 
 38R4_x    =0;                 
 39R4_y    =0;                 
 40%Vzwp    =0;                 
 41%mu_max  =0.9;
 42
 43%Wielbelastingen
 44
 45%Statische wielbelasting
 46Ws_fl = (Lf/(2*L))*M*g;
 47Ws_fr = Ws_fl;
 48Ws_rl = (Lr/(2*L))*M*g;
 49Ws_rr = Ws_rl;
 50
 51Ws_f = Ws_fl + Ws_fr;
 52Ws_r = Ws_rl + Ws_rr;
 53
 54%Laterale wielbelasting (door rollen van de wagen)
 55FfsM = (Ksf*M*ay*D)/(Tf*(Ksf+Ksr-(M*g*D)));   %Kracht door rolmoment
 56FrsM = (Ksr*M*ay*D)/(Tr*(Ksf+Ksr-(M*g*D)));
 57
 58FfsF = (Ws_f*ay*Hrcf)/Tf;                    %Kracht door afgeveerde massa
 59FrsF = (Ws_r*ay*Hrcr)/Tr;
 60
 61FfuF = (Mu_f*ay*Hurcf)/Tf;                   %Kracht door onafgeveerde massa
 62FruF = (Mu_r*ay*Hurcr)/Tr;
 63
 64Ff = FfsM + FfsF + FfuF;                    %Totale gewichtsverplaatsing
 65Fr = FrsM + FrsF + FruF;
 66
 67Wy_fl = Ff/2;                               %Kracht per wiel
 68Wy_fr = -Wy_fl;
 69Wy_rl = Fr/2;
 70Wy_rr = -Wy_rl;
 71
 72%Longitudinale wielbelasting
 73Wx_fl = (-H/(2*L))*M*ax;
 74Wx_fr = Wx_fl;
 75Wx_rl = (H/(2*L))*M*ax;
 76Wx_rr = Wx_rl;
 77
 78%Aerodynamische wielbelasting
 79LF = -0.0237*(Vzwp)^2 + 0.0958*Vzwp;                %Nog na te kijken
 80LR = -0.0237*(Vzwp)^2 + 0.0958*Vzwp;                %Nog na te kijken
 81
 82Wa_fl = (-LF/(2*L));  %+((Krf/(Krf+Krr))*(Rm/Tf));      %is er hier 2x rolmoment??? (zie lateraal)
 83Wa_fr = (-LF/(2*L));  %-((Krf/(Krf+Krr))*(Rm/Tf));
 84Wa_rl = (-LR/(2*L));  %+((Krr/(Krf+Krr))*(Rm/Tr));
 85Wa_rr = (-LR/(2*L));  %-((Krr/(Krf+Krr))*(Rm/Tr));
 86
 87%anti-roll bar wielbelasting
 88Wab_fl = (Krf_a/((Krf_a+((Kwf*(Tf)^2)/2))*Wy_fl));
 89Wab_fr = -Wab_fl;
 90Wab_rl = (Krr_a/((Krr_a+((Kwr*(Tr)^2)/2))*Wy_rl));
 91Wab_rr = -Wab_rl;
 92
 93%Totale belasting per wiel
 94Wfl = Ws_fl + Wy_fl + Wx_fl + Wa_fl + Wab_fl;   %Ongebruikt, kan later eventueel gebruikt worden voor 4WD
 95Wfr = Ws_fr + Wy_fr + Wx_fr + Wa_fr + Wab_fr;   %^ Idem
 96Wrl = Ws_rl + Wy_rl + Wx_rl + Wa_rl + Wab_rl;
 97Wrr = Ws_rr + Wy_rr + Wx_rr + Wa_rr + Wab_rr;
 98
 99
100
101%%Omrekening naar torques
102
103%Wrijvingscirkels
104Fx_rl = sqrt((Wrl^2*mu_max^2)-(Wrl^2*ay^2));
105Fx_rr = sqrt((Wrr^2*mu_max^2)-(Wrr^2*ay^2));
106
107%Maximum torques
108Trl_max = Fx_rl*Reff;
109Trr_max = Fx_rr*Reff;
