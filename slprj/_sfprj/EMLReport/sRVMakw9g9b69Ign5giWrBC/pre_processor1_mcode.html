1function Yaw_ref  = pre_processor(Vzwp,stuurhoek,mu_max,Fy_fl,Fy_fr,Fy_rl,Fy_rr,alfa_fl,alfa_fr,alfa_rl,alfa_rr)
 2%De pre-processor berekend de gewenste yaw rate door het minimum te nemen
 3%van de ideale en de maximale yaw rate waarde.
 4
 5
 6M       =265;               
 7L       =1.525;             
 8Lf      =0.780;             
 9Lr      =0.745;             
10Tf      =1.200;             
11Tr      =1.200;             
12g       =9.81;              
13Ksf     =12153;             
14Ksr     =14563;             
15Krf_a   =183.26;            
16Krr_a   =325.31;            
17Kwf     =0;                 
18Kwr     =0;                 
19Hrcf    =0.032;             
20Hrcr    =0.041;             
21Hurcf   =0;                 
22Hurcr   =0;                 
23H       =0.2735;            
24D       =0.2369;            
25Rm      =576.35;            
26Reff    =0.2325;            
27Gr      =10.32;             
28Sr      =0.600;             
29Tmax    =38.5;              
30Cte     =0;                 
31ax      =0;                 
32ay      =0;                 
33Mu_f    =8.5;               
34Mu_r    =8.5;               
35R1      =0;                 
36R2      =0;                 
37R4_x    =0;                 
38R4_y    =0;                 
39Vzwp    =0;                 
40mu_max  =0.9;
41
42X = [];
43%Berekening massa's
44Mf = M*(Lr/L);
45Mr = M*(Lf/L);
46
47%Berekening Cornering stiffness
48C_fl = (Fy_fl/alfa_fl);
49C_fr = (Fy_fr/alfa_fr);
50C_rl = (Fy_rl/alfa_rl);
51C_rr = (Fy_rr/alfa_rr);
52
53%Gemiddelde van de cornering stiffness vooraan en achteraan nemen
54Cf = ((C_fl + C_fr)/ 2);
55Cr = ((C_rl + C_rr)/ 2);
56%Berekening van de onderstuur factor.
57Ku = ((Mf/Cf)-(Mr/Cr));
58
59%Berekening van de ideale yaw rate.
60Yaw_id = ((Vzwp*stuurhoek)/((Lf+Lr)+Ku*(Vzwp)^2));
61X = [X Yaw_id];
62% berekening van de sliphoek beita
63beta = atan(Lf * tan(stuurhoek)/ (Lf+Lr)) ;
64
65%Berekening van de maximale yaw rate.
66Yaw_max = ((mu_max*g)/(Vzwp*cos(beta)))-Vzwp*diff(beta)*cos(beta);
67X = [X Yaw_max];
68%Minimale yaw rate.
69M = min(X);
70Yaw_ref = M;
